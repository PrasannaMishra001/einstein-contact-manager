# Root folder
$root = "ContactManager"

# Create root folder
New-Item -Path $root -ItemType Directory -Force

# Create files in root
New-Item -Path "$root\README.md" -ItemType File -Force
New-Item -Path "$root\requirements.txt" -ItemType File -Force
New-Item -Path "$root\main.py" -ItemType File -Force

# Create data folder and file
New-Item -Path "$root\data" -ItemType Directory -Force
New-Item -Path "$root\data\contacts.json" -ItemType File -Force

# Create contacts package folder and files
New-Item -Path "$root\contacts" -ItemType Directory -Force
New-Item -Path "$root\contacts\__init__.py" -ItemType File -Force
New-Item -Path "$root\contacts\contact.py" -ItemType File -Force
New-Item -Path "$root\contacts\addressbook.py" -ItemType File -Force
New-Item -Path "$root\contacts\fileops.py" -ItemType File -Force
New-Item -Path "$root\contacts\utils.py" -ItemType File -Force

# Create tests folder and file
New-Item -Path "$root\tests" -ItemType Directory -Force
New-Item -Path "$root\tests\test_addressbook.py" -ItemType File -Force

Write-Host "Project structure created successfully!"
